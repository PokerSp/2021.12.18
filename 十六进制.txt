#include <stdio.h>
int main ()
{
    int N=0,i=0;
    printf("请输入多少个十六进制\n");
    scanf("%d",&N);
    int C[N/3*4+4];
    int A[N];
    char B[64];
    for(i=0;i<=25;i++)
    {
        B[i]=65+i;

    }

    for(i=26;i<=51;i++)
    {

        B[i]=97+i-26;
    }
    for(i=52;i<=61;i++)
    {

        B[i]=48+i-52;
    }
    B[62]='+';
    B[63]='/';
for(i=0;i<=63;i++)
    {

        printf("%c",B[i]);
    }
    printf("请输入十六进制\n");
    for(i=0;i<N;i++)
    {
        scanf("%x",&A[i]);
    }
    if(N%3==0)
    {
        for(i=0;i<N/3;i++)
        {
          C[0+i*4]=A[i*3+0]/4;
          C[1+i*4]=A[i*3+0]%4*16+A[i*3+1]/16;
          C[2+i*4]=A[i*3+1]%16*4+A[i*3+2]/64;
          C[3+i*4]=A[i*3+2]%64;

        }

    }
    if(N%3==2)
    {
        for(i=0;i<N/3;i++)
        {
        C[0+i*4]=A[i*3+0]/4;
          C[1+i*4]=A[i*3+0]%4*16+A[i*3+1]/16;
          C[2+i*4]=A[i*3+1]%16*4+A[i*3+2]/64;
          C[3+i*4]=A[i*3+2]%64;


        }
        C[N/3*4+0]=A[N-1-1]/4;
        C[N/3*4+1]=A[N-1]/16+A[N-1-1]%4*16;
        C[N/3*4+2]=A[N-1]%16;
        C[N/3*4+3]=0;

    }
    if(N%3==1)
    {
        for(i=0;i<N/3;i++)
        {
         C[0+i*4]=A[i*3+0]/4;
          C[1+i*4]=A[i*3+0]%4*16+A[i*3+1]/16;
          C[2+i*4]=A[i*3+1]%16*4+A[i*3+2]/64;
          C[3+i*4]=A[i*3+2]%64;


        }
        C[N/3*4+0]=A[N-1]/4;
        C[N/3*4+1]=A[N-1]%4;
        C[N/3*4+2]=0;
        C[N/3*4+3]=0;

    }


      


    for(i=0;i<(N-1)/3*4+4;i++)
    {
        printf("%c ",B[C[i]]);

    }
    printf("\n");
    if(N%3!=0)
        printf("=");
}
